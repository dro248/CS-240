drop table if exists Batch;
drop table if exists Cell;
drop table if exists Field;
drop table if exists Project;
drop table if exists Record;
drop table if exists User;


CREATE TABLE Batch
(
ID integer not null primary key autoincrement,
isAvailable boolean,
filename text,
projectID int
);



CREATE TABLE Cell
(
ID integer not null primary key autoincrement,
RecordCoord int,
FieldCoord int,
parentID int,
CellValue text
);



CREATE TABLE Field
(
ID integer not null primary key autoincrement,
ParentBatchID int,
Title text,
XCoord int,
Width int,
FieldNumber int,
helpHTML text,
knownData text
);



CREATE TABLE Project
(
ID integer not null primary key autoincrement,
Title text,
RecordsPerImage int,
FirstYCoord int,
RecordHeight int
);



CREATE TABLE Record
(
ID integer not null primary key autoincrement,
RecordNumber int,
parentBatchID int
);



CREATE TABLE User
(
ID integer not null primary key autoincrement,
CurrentBatchID int,
IndexRecords int,
firstName text,
lastName text,
email text,
username text,
password text
);
